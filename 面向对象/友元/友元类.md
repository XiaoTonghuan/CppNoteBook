友元类

``` c++
class MensDormitory  //男生宿舍
{
	friend class Adminstrator;  //将管理员类设定为友元。

private:
	int m_FourMenRoom; //四人间
public:
	MensDormitory(int FourMenRoom)
	{
		m_FourMenRoom = FourMenRoom;
	}
};
class Adminstrator  //管理员类
{
public:

	MensDormitory* Numb; //管理员管理着所有宿舍。

	Adminstrator()
	{
		//创建一个管理员的时候，顺便在堆区开辟一个宿舍，给宿舍号赋初值为0。
		Numb = new MensDormitory(0); 
	}
	void visit() //传入管理员想要访问的宿舍号，直接访问。
	{
		cout << "i am adminstrator ,i will enter a room" << Numb->m_FourMenRoom;
		//在类中创建的类成员就像变量一样，可以直接在函数中访问。
	}
};
void test01()
{
	Adminstrator A1; //创建一位管理员。
	A1.visit(); //这位管理员正在访问10号宿舍楼
}

int main()
{
	test01();
}
```
