# 二进制的方式写文件
```c++
#include<iostream>
#include<ftream>
using namespace std;
class person {
public:
    char m_name[111];
    int m_age;
};
int main()
{
    ofstream ofs("person.txt",ios::out|ios::binary); //声明变量时同时调用有参构造函数。
    person p1("张三",14);  //初始化一个person对象
    ofs.write( (const char*)&p1 , sizeof(p1) )  //利用write函数，将这个对象的数据写入文件，首先要把这个对象强制转换成字符型的地址，后面写他占用的内存空间。
    ofs.close();
}
```
# 二进制的方式读文件
```c++
# include <iostream>
# include<fstream>
class person{
public:
    char m_name[109];
    int m_age;
};
int main()
{
    ifstream ifs;
    person p1; //用来存放读入的文件。
    ifs.open("person.txt",ios::in|ios::binary)
    ifs.read((char*)&p1,sizeof(person)); //将文件的地址转化成字符。以及占用的内存空间。
    ifs.close();
}
```
