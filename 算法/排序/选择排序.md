选择排序

# 算法介绍
从左向右依次寻找最小的数字，将其放在最左边，去掉最左边，重复上述过程，直至成为有序数列。
#代码实现
``` c++
#include<iostream>
using namespace std;
void SelectSort (int arry[] ,int len ) //传入数组和数组的长度。
{
	 int min = 0 ; //设定一个最小值。
	for(int i = 0 ;i < len ; i++ )
	{
		min =i ; //假设最小值为第i位。
		for (int j=i+1 ; j < len ; j++ ) //遍历最小值之后的数据
		{
			if( arry[j] < arry[min] ) //判断最小的数据
			{
				min = j ; //标记最小数据的下标
			}
		}
		int temp = arry[i] ;  //交换最小数据和原数据
		arry[i] = arry[min] ;
		arry[min] = temp ;
		
	}
 } 

int main()
{
	cout<<"未排序的数组为"<<endl; 
	int arry[10] = {8,3,56,444,123,589,2541,36,251,1024} ;
	int len = sizeof(arry) / sizeof (arry[0]) ;
	for ( int i = 0 ; i<len ; i++ )
	{
		cout<<arry[i]<<" ";
	}
	cout<<endl;
	SelectSort (arry,len) ;
	cout<<"排序之后的数组为"<<endl;
	 for ( int i = 0 ; i<len ; i++ )
	{
		cout<<arry[i]<<" ";
	}
}
```
