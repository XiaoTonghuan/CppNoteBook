# 加法

```c++
#include<iostream>
#include<string>
using std::cin;
using std::cout;
const int maxn = 1000;  //定义常量
struct bigInteger
{
    int len;
    int v[maxn]{0}; //初始化
};
bigInteger read()  //读取一个数
{
    bigInteger a;
    std::string str;
    cin>>str;
    int n = str.size();
    for(int i = str.size() - 1 ;  i >= 0 ; --i) 
        a.v[n - 1 - i] = str[i] - '0';  //反向储存字符串的数据
    a.len = str.size(); //确定长度
    return a;
}
void output_bigInteger(bigInteger& a) //打印一个大数
{
    for(int i = a.len - 1; i >= 0 ; --i)
        cout<<a.v[i];
    cout<<std::endl;
}
bigInteger add_bigInteger(bigInteger &a,bigInteger &b)
{
    bigInteger c;
    int x = 0;  //进位
    int i = 0;  //位数
    while(i<a.len || i<b.len)
    {
        c.v[i] = a.v[i] + b.v[i] + x;  //进位+a,b的该位
        x = c.v[i] / 10;  //确定进位
        c.v[i] %= 10;  //确定该位
        ++i; //位数++
    }
    if(x == 0) i--; //位数最终多增加一次，如果最后进位是0，则去掉可能存在的最高位
    else c.v[i] = x;  //否则最到位==进位
    c.len = i + 1;  //长度== 位数（下标从0开始） + 1
    return c;
}
int main()
{
    bigInteger a,b;
    a = read();
    b = read();
    bigInteger c = add_bigInteger(a,b);
    output_bigInteger(c);
    system("pause");
}
```
