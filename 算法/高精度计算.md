```c++
vector<int> add(vector<int>& A , vector<int>& B){
    vector<int> res;
    int t = 0; //表示进位
    for(int i = 0 ; i < A.size() || i < B.size(); ++i){
        if( i < A.size() ) t += A[i];
        if( i < B.size() ) t += B[i];
        res.push_back(t % 10);
        t-=10; 
    }
    if(t>0) res.push_back(1);
    return res;
}
```

```c++
vector<int> mul(vector<int>& A , int &B){
    vector<int> C;
    int t = 0; 
    for(int i = 0 ; i < A.size() ; ++i){
        // 1234
        //   98
        
        t += A[i] * B;  
        // 4 * 98 --- 392

        C.push_back(t % 10);
        // 写2

        t /= 10;
        //进39

    }
    while(t) C.push_back(t%10), t/=10;
    // 一直进位
    while(C.size() > 1 && C.back() == 0 ) C.pop_back();
    //去前导0
	return C;
}
```

# 除法

```c++
//除法是先算出高位，和+-*不一样，因此要reverse一下
vector<int> div(vector<int>& A , int &B  ,int &r){  //求余数
    vector<int> C;
    r = 0 ;
    for(int i = 0 ; i < A.size() ; ++i){ // 从高位到低位枚举
        r = r * 10 + A[i];  
        //1234
        //  11
        //0*10+1 / 11 ...
        C.push_back(r/B);
        // 商
        r %= B;
        //准备下一位
    }
    while(C.size()>1 && C.back()==0) C.pop_back();  
    //删除前导0
    //0122 --- 122
    reverse(C.begin(), C.end());  // 让其变成221 输出时才是122
    return C;
}
```
